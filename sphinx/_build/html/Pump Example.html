Pump¶

Simulate Pump for costing purposes (no pressure input necessary)¶

In [1]:

from biosteam import Species, Stream, Pump
Stream.species = Species('Water', 'Ethanol')

# Simulate pump
feed = Stream('feed', Water=200, T=350)
P1 = Pump('P1', ins=feed, outs='out')
P1.simulate()

# Show results
P1.diagram()
P1.show()
