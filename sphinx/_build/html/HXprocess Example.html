HXprocess¶

Sensible fluids case:¶

In [1]:

from biosteam import Species, Stream, HXprocess
Stream.species = Species('Water', 'Ethanol')

# Simulate counter-current heat exchanger
in1 = Stream('in1', Water=200, T=350)
in2 = Stream('in2', Ethanol=200)
hx = HXprocess('hx', ins=(in1, in2), outs=('out1', 'out2'), Type='ss')
hx.simulate()

# Show results
hx.diagram()
hx.show()
