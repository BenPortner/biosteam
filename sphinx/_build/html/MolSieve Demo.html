Molecular Sieves¶

Based on bioethanol plant molecular sieves. Cost based on scaling factor from NREL report.¶

In [1]:

from biosteam import Species, Units, Stream

# Set up stream
Stream.species = Species('Water', 'Ethanol')
stream = Stream(flow=(75.7, 286), phase='g')
stream.T = stream.bubble_point()[0]

# Set up column
MS1 = Units.MolSieve('MS1', split=(0.160, 0.925))
MS1.ins = stream

# Run all methods
MS1.run()
MS1.operation()
MS1.design()
MS1.cost()

# See all results
MS1.diagram
MS1.show(T='degC', P='atm', in_fractions= True)
print('\nresults...\n' + str(MS1.results) + '\n')
print('power utility...')
MS1.power_utility.show(Cost='USD/yr')
